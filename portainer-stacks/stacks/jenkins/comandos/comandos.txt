// Cada comando deve ser adicionado como pipeline no Jenkins.


-- Camada Bronze

pipeline {
    agent any

    stages {

        stage('Executar Notebook Bronze') {
            steps {
                sh '''
                docker exec jupyter \
                /opt/conda/bin/jupyter nbconvert \
                --to notebook \
                --execute \
                --inplace \
                /home/jovyan/work/01_extracao_bronze.ipynb
                '''
            }
        }

    }
}
